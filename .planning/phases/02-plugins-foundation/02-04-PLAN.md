{

    var kernelBuilder = Kernel.CreateBuilder();
    services.AddSingleton(kernelBuilder.Build());

    // Register HttpClient

    // Register services

## Plugin Registration Pattern (from RESEARCH.md)
Step 1 - DI Registration (in AddFlightAgentPlugins):
```csharp
services.AddSingleton<IAirportPlugin, AirportPlugin>();
services.AddSingleton<IWeatherPlugin, WeatherPlugin>();
```

Step 2 - SK Registration (in Program.cs after app.Build()):
```csharp
using (var scope = app.Services.CreateScope())
{
    var sp = scope.ServiceProvider;
    var kernel = sp.GetRequiredService<Kernel>();
    
    // SK reflects on concrete class, so resolve concrete type
    var airportPlugin = sp.GetRequiredService<AirportPlugin>();
    kernel.Plugins.AddFromObject(airportPlugin, "Airport");
    
    var weatherPlugin = sp.GetRequiredService<WeatherPlugin>();
    kernel.Plugins.AddFromObject(weatherPlugin, "Weather");
}
```

using Microsoft.Extensions.DependencyInjection;
using Microsoft.Extensions.Configuration;
using Microsoft.SemanticKernel;
using FlightAgent.Core.Interfaces;

namespace FlightAgent.Infrastructure;

public static class DependencyInjection
{
    public static IServiceCollection AddInfrastructure(this IServiceCollection services, IConfiguration configuration)
    {
        // Configure Semantic Kernel
        var kernelBuilder = Kernel.CreateBuilder();

        // Add AI service (configure with your preferred provider)
        // Example with Azure OpenAI:
        // kernelBuilder.AddAzureOpenAIChatCompletion(
        //     deploymentName: configuration["AzureOpenAI:DeploymentName"]!,
        //     endpoint: configuration["AzureOpenAI:Endpoint"]!,
        //     apiKey: configuration["AzureOpenAI:ApiKey"]!);

        // Example with OpenAI:
        // kernelBuilder.AddOpenAIChatCompletion(
        //     modelId: configuration["OpenAI:ModelId"] ?? "gpt-4",
        //     apiKey: configuration["OpenAI:ApiKey"]!);

        services.AddSingleton(kernelBuilder.Build());

        // Register HttpClient for health checks and external API calls
        services.AddHttpClient();

        // Register services
        services.AddScoped<IFlightSearchService, Services.SemanticFlightSearchService>();
        services.AddScoped<IBookingService, Services.SemanticBookingService>();

        // Register health checks with custom external API check
        services.AddHealthChecks()
            .AddCheck<Health.ExternalApiHealthCheck>("external_api", tags: new[] { "external" });

        return services;
    }

    /// <summary>
    /// Adds FlightAgent plugins to the DI container.
    /// </summary>
    public static IServiceCollection AddFlightAgentPlugins(this IServiceCollection services)
    {
        // Per D-26: Step 1 - Register plugin classes in DI
        // Singleton because plugins hold state (airport data dictionary, cache)
        services.AddSingleton<IAirportPlugin, Plugins.AirportPlugin>();
        services.AddSingleton<IWeatherPlugin, Plugins.WeatherPlugin>();

        // Ensure IMemoryCache is available (should already be registered in AddInfrastructure via AddHttpClient)
        // But we explicitly ensure it's here for WeatherPlugin
        services.AddMemoryCache();

        return services;
    }
}
```

// Per D-26, D-28: Step 2 - Register plugins with Semantic Kernel
// This must happen after app.Build() because we need the service provider
using (var scope = app.Services.CreateScope())
{
    var sp = scope.ServiceProvider;
    var kernel = sp.GetRequiredService<Kernel>();

    // Per D-28: Resolve concrete type from DI, not interface
    // SK reflects on concrete class methods to find [KernelFunction] attributes
    var airportPlugin = sp.GetRequiredService<AirportPlugin>();
    kernel.Plugins.AddFromObject(airportPlugin, "Airport");

    var weatherPlugin = sp.GetRequiredService<WeatherPlugin>();
    kernel.Plugins.AddFromObject(weatherPlugin, "Weather");
}

app.Run();
```

using FluentAssertions;
using Microsoft.Extensions.DependencyInjection;
using FlightAgent.Core.Interfaces;

public class PluginRegistrationTests

    [Fact]

    {
        // Arrange
        var services = new ServiceCollection();
        services.AddMemoryCache();
        services.AddHttpClient();

        var plugin = provider.GetService<IAirportPlugin>();

        var plugin = provider.GetService<IWeatherPlugin>();
