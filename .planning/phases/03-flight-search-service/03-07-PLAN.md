using FluentAssertions;
using Microsoft.Extensions.Configuration;
using Moq;
using Moq.Protected;
using System.Net;
using System.Text.Json;
using Xunit;
using FlightAgent.Infrastructure.Models;
using FlightAgent.Infrastructure.Services;

namespace FlightAgent.Infrastructure.Tests.Services;

public class AviationStackServiceTests
{
    [Fact]
    public async Task GetFlightStatusAsync_ReturnsFlightInfo_ForValidFlight()
    {
        // Arrange
        var mockResponse = CreateMockResponse(new FlightData
        {
            FlightDate = "2026-04-17",
            FlightStatus = "active",
            Departure = new AirportTimes
            {
                Iata = "JFK",
                Airport = "John F. Kennedy International",
                Delay = 15,
                Scheduled = "2026-04-17T08:00:00+00:00",
                Actual = "2026-04-17T08:15:00+00:00"
            },
            Arrival = new AirportTimes
            {
                Iata = "LAX",
                Airport = "Los Angeles International",
                Delay = 0,
                Scheduled = "2026-04-17T11:30:00+00:00"
            },
            Airline = new AirlineInfo { Name = "American Airlines", Iata = "AA" },
            Flight = new FlightIdentifier { Number = "1004", Iata = "AA1004" }
        });

        var service = CreateService(mockResponse);

        // Act
        var result = await service.GetFlightStatusAsync("AA1004", "2026-04-17");

        // Assert
        result.Should().NotBeNull();
        result!.FlightNumber.Should().Be("AA1004");
        result.Airline.Should().Be("American Airlines");
        result.DepartureAirport.Should().Be("JFK");
        result.ArrivalAirport.Should().Be("LAX");
        result.Status.Should().Be("on_time");
        result.DelayMinutes.Should().Be(15);
        result.ScheduledDeparture.Should().BeCloseTo(new DateTime(2026, 4, 17, 8, 0, 0), TimeSpan.FromMinutes(1));
        result.ActualDeparture.Should().NotBeNull();
    }

    [Fact]
    public async Task GetFlightStatusAsync_ReturnsNull_ForEmptyResponse()
    {
        // Arrange
        var mockResponse = CreateMockResponse(new List<FlightData>());
        var service = CreateService(mockResponse);

        // Act
        var result = await service.GetFlightStatusAsync("AA1004", "2026-04-17");

        // Assert
        result.Should().BeNull();
    }

    [Fact]
    public async Task GetFlightStatusAsync_ReturnsNull_ForNullInputs()
    {
        // Arrange
        var service = CreateService(CreateMockResponse(new FlightData()));

        // Act & Assert
        (await service.GetFlightStatusAsync(null!, "2026-04-17")).Should().BeNull();
        (await service.GetFlightStatusAsync("AA1004", null!)).Should().BeNull();
        (await service.GetFlightStatusAsync("", "2026-04-17")).Should().BeNull();
        (await service.GetFlightStatusAsync("AA1004", "")).Should().BeNull();
    }

    [Theory]
    [InlineData("active", "on_time")]
    [InlineData("scheduled", "on_time")]
    [InlineData("landed", "on_time")]
    [InlineData("delayed", "delayed")]
    [InlineData("cancelled", "cancelled")]
    [InlineData("unknown_status", "unknown")]
    public async Task GetFlightStatusAsync_MapsStatus_Correctly(string apiStatus, string expectedStatus)
    {
        // Arrange
        var flightData = CreateFlightDataWithStatus(apiStatus);
        var mockResponse = CreateMockResponse(flightData);
        var service = CreateService(mockResponse);

        // Act
        var result = await service.GetFlightStatusAsync("AA1004", "2026-04-17");

        // Assert
        result!.Status.Should().Be(expectedStatus);
    }

    private static Mock<HttpMessageHandler> CreateMockResponse(FlightData data)
    {
        var response = new FlightData[] { data };
        var json = JsonSerializer.Serialize(new AviationStackResponse { Data = response.ToList() });

        var mockHandler = new Mock<HttpMessageHandler>();
        mockHandler.Protected()
            .Setup<Task<HttpResponseMessage>>(
                "SendAsync",
                ItExpr.IsAny<HttpRequestMessage>(),
                ItExpr.IsAny<CancellationToken>())
            .ReturnsAsync(new HttpResponseMessage(HttpStatusCode.OK)
            {
                Content = new StringContent(json)
            });

        return mockHandler;
    }

    private static Mock<HttpMessageHandler> CreateMockResponse(List<FlightData> data)

        Flight = new FlightIdentifier { Number = "1004", Iata = "AA1004" }
    };
}

using FluentAssertions;
using Microsoft.SemanticKernel;
using Moq;
using Xunit;
using FlightAgent.Core.Interfaces;
using FlightAgent.Core.Models;
using FlightAgent.Infrastructure.Plugins;

namespace FlightAgent.Infrastructure.Tests.Plugins;

public class FlightPluginTests
{
    [Fact]
    public async Task GetFlightStatusAsync_ReturnsFlightInfo_FromService()
    {
        // Arrange
        var expectedFlight = new FlightInfo(
            FlightNumber: "AA1004",
            Airline: "American Airlines",
            DepartureAirport: "JFK",
            ArrivalAirport: "LAX",
            ScheduledDeparture: new DateTime(2026, 4, 17, 8, 0, 0),
            ActualDeparture: new DateTime(2026, 4, 17, 8, 15, 0),
            Status: "on_time",
            DelayMinutes: 15
        );

        var mockService = new Mock<IAviationStackService>();
        mockService.Setup(s => s.GetFlightStatusAsync("AA1004", "2026-04-17", default))
            .ReturnsAsync(expectedFlight);

        var plugin = new FlightPlugin(mockService.Object);

        // Act
        var result = await plugin.GetFlightStatusAsync("AA1004", "2026-04-17");

        // Assert
        result.Should().NotBeNull();
        result!.FlightNumber.Should().Be("AA1004");
        result.Airline.Should().Be("American Airlines");
        result.Status.Should().Be("on_time");
    }

    [Fact]
    public async Task GetFlightStatusAsync_ReturnsNull_ForNullInputs()
    {
        // Arrange
        var mockService = new Mock<IAviationStackService>();
        var plugin = new FlightPlugin(mockService.Object);

        // Act & Assert
        (await plugin.GetFlightStatusAsync(null!, "2026-04-17")).Should().BeNull();
        (await plugin.GetFlightStatusAsync("AA1004", null!)).Should().BeNull();
        (await plugin.GetFlightStatusAsync("", "2026-04-17")).Should().BeNull();
        (await plugin.GetFlightStatusAsync("AA1004", "")).Should().BeNull();
    }

    [Fact]
    public async Task GetFlightStatusAsync_ReturnsNull_WhenServiceReturnsNull()
    {
        // Arrange
        var mockService = new Mock<IAviationStackService>();
        mockService.Setup(s => s.GetFlightStatusAsync(It.IsAny<string>(), It.IsAny<string>(), default))
            .ReturnsAsync((FlightInfo?)null);

        var plugin = new FlightPlugin(mockService.Object);

        // Act
        var result = await plugin.GetFlightStatusAsync("XX9999", "2026-04-17");

        // Assert
        result.Should().BeNull();
    }

    [Fact]
    public void KernelFunctionAttribute_IsGetFlightStatus()
    {
        // Arrange
        var method = typeof(FlightPlugin).GetMethod("GetFlightStatusAsync");

        // Assert
        method.Should().NotBeNull();
        var kernelFunction = method!.GetCustomAttributes(typeof(KernelFunctionAttribute), false)
            .FirstOrDefault() as KernelFunctionAttribute;
        kernelFunction.Should().NotBeNull();
        kernelFunction!.Name.Should().Be("get_flight_status");
    }
}
