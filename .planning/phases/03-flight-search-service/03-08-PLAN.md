using FluentAssertions;
using Microsoft.Extensions.Http.Resilience;
using Microsoft.Extensions.DependencyInjection;
using Microsoft.Extensions.Configuration;
using Moq;
using Xunit;
using FlightAgent.Infrastructure.Services;
using System.Net;
using System.Text.Json;
using FlightAgent.Infrastructure.Models;

namespace FlightAgent.Infrastructure.Tests.Services;

public class ResilienceTests
{
    [Fact]
    public void HttpClientServiceCollection_HasResiliencePipeline()
    {
        // Arrange
        var services = new ServiceCollection();
        var mockConfig = new Mock<IConfiguration>();
        mockConfig.Setup(c => c["AviationStack:ApiKey"]).Returns("test-key");

        // Act
        services.AddHttpClient<IAviationStackService, AviationStackService>(client =>
        {
            client.BaseAddress = new Uri("https://api.aviationstack.com/v1/");
        })
        .AddResilienceHandler("AviationStackResilience", builder =>
        {
            builder.AddTimeout(new HttpTimeoutStrategyOptions { Timeout = TimeSpan.FromSeconds(8) });
            builder.AddRetry(new HttpRetryStrategyOptions
            {
                MaxRetryAttempts = 3,
                BackoffType = DelayBackoffType.Exponential,
                UseJitter = true,
                ShouldHandle = new PredicateBuilder<HttpResponseMessage>()
                    .Handle<HttpRequestException>()
                    .HandleResult(r => !r.IsSuccessStatusCode)
            });
            builder.AddCircuitBreaker(new HttpCircuitBreakerStrategyOptions
            {
                FailureRatio = 0.5,
                SamplingDuration = TimeSpan.FromSeconds(30),
                MinimumThroughput = 5,
                BreakDuration = TimeSpan.FromSeconds(30)
            });
        });

        var provider = services.BuildServiceProvider();

        // Assert
        var httpClientFactory = provider.GetService<IHttpClientFactory>();
        httpClientFactory.Should().NotBeNull();
    }

    [Fact]
    public void RetryStrategy_UsesExponentialBackoff()
    {
        // Arrange
        var retryOptions = new HttpRetryStrategyOptions
        {
            MaxRetryAttempts = 3,
            BackoffType = DelayBackoffType.Exponential,
            UseJitter = true
        };

        // Assert
        retryOptions.MaxRetryAttempts.Should().Be(3);
        retryOptions.BackoffType.Should().Be(DelayBackoffType.Exponential);
        retryOptions.UseJitter.Should().BeTrue();
    }

    [Fact]
    public void CircuitBreakerStrategy_HasCorrectThresholds()
    {
        // Arrange
        var cbOptions = new HttpCircuitBreakerStrategyOptions
        {
            FailureRatio = 0.5,
            SamplingDuration = TimeSpan.FromSeconds(30),
            MinimumThroughput = 5,
            BreakDuration = TimeSpan.FromSeconds(30)
        };

        // Assert
        cbOptions.FailureRatio.Should().Be(0.5);
        cbOptions.SamplingDuration.Should().Be(TimeSpan.FromSeconds(30));
        cbOptions.MinimumThroughput.Should().Be(5);
        cbOptions.BreakDuration.Should().Be(TimeSpan.FromSeconds(30));
    }

    [Fact]
    public void TimeoutStrategy_Has8SecondTimeout()
    {
        // Arrange
        var timeoutOptions = new HttpTimeoutStrategyOptions
        {
            Timeout = TimeSpan.FromSeconds(8)
        };

        // Assert
        timeoutOptions.Timeout.Should().Be(TimeSpan.FromSeconds(8));
    }

    [Fact]
    public async Task AviationStackService_Retries_OnServerError()
    {
        // Arrange
        var callCount = 0;
        var mockHandler = new Mock<HttpMessageHandler>();
        mockHandler.Protected()
            .Setup<Task<HttpResponseMessage>>(
                "SendAsync",
                ItExpr.IsAny<HttpRequestMessage>(),
                ItExpr.IsAny<CancellationToken>())
            .ReturnsAsync(() =>
            {
                callCount++;
                if (callCount < 3)
                    return new HttpResponseMessage(HttpStatusCode.InternalServerError);
                return new HttpResponseMessage(HttpStatusCode.OK)
                {
                    Content = new StringContent(CreateValidJsonResponse())
                };
            });

        var httpClient = new HttpClient(mockHandler.Object);
        httpClient.BaseAddress = new Uri("https://api.aviationstack.com/v1/");

        var mockConfig = new Mock<IConfiguration>();
        mockConfig.Setup(c => c["AviationStack:ApiKey"]).Returns("test-key");

        var service = new AviationStackService(httpClient, mockConfig.Object);

        // Act
        var result = await service.GetFlightStatusAsync("AA1004", "2026-04-17");

        // Assert
        callCount.Should().Be(3, "Should have retried 3 times before success");
        result.Should().NotBeNull();
    }

    [Fact]
    public async Task AviationStackService_ReturnsNull_OnNonRetryableError()
    {
        // Arrange
        var mockHandler = new Mock<HttpMessageHandler>();
        mockHandler.Protected()
            .Setup<Task<HttpResponseMessage>>(
                "SendAsync",
                ItExpr.IsAny<HttpRequestMessage>(),
                ItExpr.IsAny<CancellationToken>())
            .ReturnsAsync(new HttpResponseMessage(HttpStatusCode.BadRequest)
            {
                Content = new StringContent("Bad request")
            });

        var httpClient = new HttpClient(mockHandler.Object);
        httpClient.BaseAddress = new Uri("https://api.aviationstack.com/v1/");

        var mockConfig = new Mock<IConfiguration>();
        mockConfig.Setup(c => c["AviationStack:ApiKey"]).Returns("test-key");

        var service = new AviationStackService(httpClient, mockConfig.Object);

        // Act
        var result = service.Invoking(s => s.GetFlightStatusAsync("AA1004", "2026-04-17"));

        // Assert
        // BadRequest throws exception - service should handle gracefully
        await result.Should().ThrowAsync<HttpRequestException>();
    }

    private static string CreateValidJsonResponse() => JsonSerializer.Serialize(
        new AviationStackResponse
        {
            Data = new List<FlightData>
            {
                new()
                {
                    FlightDate = "2026-04-17",
                    FlightStatus = "active",
                    Departure = new AirportTimes { Iata = "JFK", Scheduled = "2026-04-17T08:00:00+00:00" },
                    Arrival = new AirportTimes { Iata = "LAX", Scheduled = "2026-04-17T11:30:00+00:00" },
                    Airline = new AirlineInfo { Name = "American Airlines", Iata = "AA" },
                    Flight = new FlightIdentifier { Number = "1004", Iata = "AA1004" }
                }
            }
        });
}
